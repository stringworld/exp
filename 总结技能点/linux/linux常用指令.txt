启动start、暂停stop、重启restart

启动mysql服务：service mysqld start
启动apache服务：service httpd start


.	/当前目录
..	/上一级目录
cd	进出目录
mkdir	创建目录
rmdir	删除目录
	-p	当子目录被删除后使它也成为空目录的话，则顺便一并删除。

cp[from][to]	复制
	Cr	连同源文件中的子目录一同拷贝

rm	删除目录或文件
	-f	文件名（目录名）：这个操作可以进行强制删除。
	-i	文件
	-r	连同这个目录下面的子目录都删除

mv[from][to]	移动目录或文件
	重命名时可以用以下方法：mv 原文件名（目录） 新的文件名（目录）。

pwd	目前所在的工作目录的绝对路径名称
clear	清除屏幕信息

chmod 权限 文件
	chmod 777 Config_Db.php

# head -n 1 /etc/issue   # 查看操作系统版本
# cat /proc/cpuinfo      # 查看CPU信息
# hostname               # 查看计算机名
# lspci -tv              # 列出所有PCI设备
# lsusb -tv              # 列出所有USB设备
# lsmod                  # 列出加载的内核模块

资源

# free -m                # 查看内存使用量和交换区使用量
# df -h                  # 查看各分区使用情况
# du -sh <目录名>        # 查看指定目录的大小
# grep MemTotal /proc/meminfo   # 查看内存总量
# grep MemFree /proc/meminfo    # 查看空闲内存量
# uptime                 # 查看系统运行时间、用户数、负载
# cat /proc/loadavg      # 查看系统负载

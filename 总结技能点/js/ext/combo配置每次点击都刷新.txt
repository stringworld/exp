onSelect:function(record,index){
	 if(this.fireEvent('beforeselect', this, record, index) !== false){
		Ext.getCmp("goods_id_has").setValue(record.data.product_id);
		Ext.getCmp("gname_has").setValue(record.data.product_name);
		Ext.getCmp("goods_no_has").setValue(record.data.product_code);
		Ext.getCmp("in_price_has").setValue(record.data.price);
		Ext.getCmp("old_num").setValue(record.data.num);
		this.collapse();
	 }
},
listeners:{
	'beforequery': function(event){
		delete event.combo.lastQuery;
	}
}

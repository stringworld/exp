if exists(select * from sysobjects where name='a')
drop view a
go
create view a as
select convert(varchar(12),convert(numeric(12,10),rand())) as num
go
if exists(select * from sysobjects where name='ran')
drop function ran
go
create function ran() returns varchar(19) as
begin
	declare @num varchar(19),@i varchar(12)
	select @i=num from a
	select @num='1010 3576 '+substring(@i,4,4)+' '+substring(@i,8,4)
	return @num
end
go
select dbo.ran()
